function suspendSchedule(scheduleId) {
    $.ajax({
        url: 'modal_order_suspend.php',
        type: 'GET',
        data: { schedule_id: scheduleId },
        success: function(response) {
            // Insertar el contenido dinámico en el contenedor de modales
            $('#dynamicModalContainer').html(response);
            // Crear y mostrar el modal de suspensión
            var suspendModal = new bootstrap.Modal(document.getElementById('suspendScheduleModal'));
            suspendModal.show();
        },
        error: function() {
            alert('Error loading the suspension modal.');
        }
    });
}
